// ═══════════════════════════════════════════════════ // MERLINSMKT CRM — Google Apps Script // Sheet: 1QiyygKV3ihMnt1RCs92kNkPtvIKzjqpKIEIWJIJgnoE // Recibe: cotizacion | lead | agenda desde el catálogo // ═══════════════════════════════════════════════════ const SHEET_ID = '1QiyygKV3ihMnt1RCs92kNkPtvIKzjqpKIEIWJIJgnoE'; const EMAIL_AVISO = 'contacto@merlinsmkt.com'; // ← cambia si quieres const FORM_OC = 'https://forms.gle/azobgzHb7zp1HsrB7'; function doPost(e) { try { const d = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); if (d.tipo === 'cotizacion') guardarCotizacion(ss, d); else if (d.tipo === 'lead') guardarLead(ss, d); else if (d.tipo === 'agenda') agendarCalendar(d, ss); return ok(); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,err:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function ok() { return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } // ── COTIZACIONES (Hoja 1 existente) ────────────── function guardarCotizacion(ss, d) { const sh = ss.getSheets()[0]; // Usa tu hoja existente if (sh.getLastRow() === 0 || sh.getRange(1,1).getValue() === '') { sh.getRange(1,1,1,10).setValues([[ 'Empresa','Servicios','Cotización N°','Correo', 'OC','Enviado','X Enviar','Teléfono','Total','Fecha' ]]); formatHeader(sh, 10); } sh.appendRow([ d.empresa, d.servicios, d.num, d.email, 'Pendiente ⏳', '', '✅ Enviar', d.telefono, d.total, d.fecha ]); notificar('🧾 Cotización ' + d.num + ' — ' + d.empresa, 'Empresa: ' + d.empresa + '\nEmail: ' + d.email + '\nServicios: ' + d.servicios + '\nTotal: ' + d.total + '\nVer: https://docs.google.com/spreadsheets/d/' + SHEET_ID); } // ── LEADS (crea hoja "Leads" si no existe) ──────── function guardarLead(ss, d) { let sh = ss.getSheetByName('Leads'); if (!sh) { sh = ss.insertSheet('Leads'); sh.getRange(1,1,1,9).setValues([[ 'Fecha','Nombre','Email','Teléfono', 'Empresa','Servicio Interés','Mensaje','Estado','Seguimiento' ]]); formatHeader(sh, 9); } sh.appendRow([ d.fecha, d.nombre, d.email, d.telefono, d.empresa||'', d.servicio||'', d.mensaje||'', '🟢 Nuevo', '' ]); notificar('🟢 Lead: ' + d.nombre, 'Nombre: ' + d.nombre + '\nEmail: ' + d.email + '\nTel: ' + d.telefono + '\nEmpresa: ' + (d.empresa||'-') + '\nInterés: ' + (d.servicio||'-') + '\nVer: https://docs.google.com/spreadsheets/d/' + SHEET_ID); } // ── AGENDA CALENDAR + hoja Agendamientos ───────── function agendarCalendar(d, ss) { const cal = CalendarApp.getDefaultCalendar(); const inicio = new Date(d.fechaReunion); const fin = new Date(inicio.getTime() + 60*60*1000); cal.createEvent('📋 MerlinsMKT — ' + d.empresa, inicio, fin, { description: '👤 ' + d.nombre + '\n' + '🏢 ' + d.empresa + '\n' + '📧 ' + d.email + '\n' + '📞 ' + d.telefono + '\n' + '💼 ' + d.servicio + '\n\n' + '📝 ' + (d.mensaje||'') + '\n\n' + '📋 OC/Contrato: ' + FORM_OC, guests: d.email, sendInvites: true }); let sh = ss.getSheetByName('Agendamientos'); if (!sh) { sh = ss.insertSheet('Agendamientos'); sh.getRange(1,1,1,7).setValues([[ 'Fecha Registro','Fecha Reunión','Nombre','Empresa', 'Email','Teléfono','Servicio' ]]); formatHeader(sh, 7); } sh.appendRow([ new Date().toLocaleDateString('es-CL'), d.fechaReunion, d.nombre, d.empresa, d.email, d.telefono, d.servicio ]); notificar('📅 Reunión: ' + d.empresa, 'Fecha: ' + d.fechaReunion + '\nContacto: ' + d.nombre + '\nEmail: ' + d.email); } function formatHeader(sh, cols) { sh.getRange(1,1,1,cols) .setFontWeight('bold') .setBackground('#0c1929') .setFontColor('#c9a44a'); } function notificar(asunto, cuerpo) { try { MailApp.sendEmail(EMAIL_AVISO, asunto, cuerpo); } catch(e) {} }
